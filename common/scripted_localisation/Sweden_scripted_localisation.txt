defined_text = { # Country
	name = swedishBudgetAmount
	text = {
		trigger = {
			check_variable = {
				var = swe_var_budget
				value = 0
				compare = less_than_or_equals
			}
		}
		localization_key = swedishBudgetAmount_zero
	}
	text = {
		trigger = {
			check_variable = {
				var = swe_var_budget
				value = 0
				compare = greater_than
			}
		}
		localization_key = swedishBudgetAmount_positive
	}
}
