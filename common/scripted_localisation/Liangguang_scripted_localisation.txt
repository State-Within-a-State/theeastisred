defined_text = {
	name = GetTangStatus
	text = {
		trigger = {
			YUN = {
				NOT = {
					has_country_leader = {
						name = "Tang Jiyao"
						ruling_only = yes
					}
				}
			}
		}
		localization_key = GXC_tang_ded
	}
	text = {
		trigger = {
			OR = {
				YUN = {
					has_country_flag = not_attending_shantou_conference
				}
				AND = {
					YUN = {
						has_country_leader = {
							name = "Tang Jiyao"
							ruling_only = yes
						}
						NOT = {
							has_country_flag = attending_shantou_conference
						}
					}
				}
			}
		}
		localization_key = GXC_tang_jerk
	}
	text = {

	}
}
