defined_text = {
	name = GetSIAprefixTH
	text = {
		trigger = {
			NOT = {
				has_cosmetic_tag = SIA_REP
				has_cosmetic_tag = THI_REP
				has_socialist_government = yes
			}
		}
		localization_key = SIA_prefix_roy_th
	}
	text = {
		trigger = {
			OR = {
				has_cosmetic_tag = SIA_REP
				has_cosmetic_tag = THI_REP
				has_socialist_government = yes
			}
		}
		localization_key = SIA_prefix_rep_th
	}
}
