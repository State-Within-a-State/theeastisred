	OR = {
		has_government = social_democrat
		has_government = social_liberal
		AND = {
			has_event_target = UPC_leader
			event_target:UPC_leader = { tag = GXC }
		}
	}
	NOT = {
		has_country_leader = {
			name = "Chen Mingshu"
			ruling_only = yes
		}
	}
}
