			original_tag = GXC
			original_tag = HNN
			original_tag = LEP
			original_tag = QIE
			original_tag = SHX
			original_tag = SIK
			original_tag = SQI
			original_tag = SZC
			original_tag = XSM
			original_tag = YUN
			original_tag = MAN
			original_tag = PRC
			original_tag = QIO
		}
	}
}
is_non_han_chinese_tag = {
	OR = {
		tag = TAN
		tag = ETS
		tag = TIB
		tag = MON
		tag = TAI
	}
}

is_lec_core_chinese_tag = {
	AND = {
	original_tag = LEC
		LEC = {
			OR = { 
			has_country_flag = LEC_Triad_Power
			has_country_flag = LEC_Shuntian_Power
			has_country_flag = shanghai_revolution
			}
		}
	}
}
is_chinese_tag = {
	custom_trigger_tooltip = {
		tooltip = is_chinese_tag
		OR = {
			is_han_chinese_tag = yes
			is_non_han_chinese_tag = yes
			tag = LEC
		}
	}
}
is_tag_in_china_area = {
	OR = {
		is_chinese_tag = yes
		original_tag = MON
		original_tag = KUM
		original_tag = TIB
		original_tag = ETS
		original_tag = LEC
		original_tag = TAI
	}
}

is_uighur_tag = {
	custom_trigger_tooltip = {
		tooltip = is_uighur_tag
		OR = {
			tag = ETS
			AND = {
				tag = TAN
				NOT = { has_cosmetic_tag = TAN_HAM }
				NOT = { has_cosmetic_tag = TAN_zhongying }
			}
		}
	}
}
is_japanese_tag = {
	OR = {
		tag = JAP
		tag = JSR
		AND = {
			tag = TRM
			has_country_flag = TRM_uses_japanese
		}
	}
}

is_mongolian_tag = {
	OR = {
		original_tag = MON
		AND = {
			tag = BRY
			NOT = {
				has_government = totalist
				has_government = paternal_autocrat
			}
		}
	}
}
