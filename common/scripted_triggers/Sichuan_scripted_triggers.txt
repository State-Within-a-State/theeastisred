SZC_agri_branch_unlocked = {
	if = {
		limit = { has_idea = SZC_famine_idea }
		NOT = { has_idea = SZC_famine_idea }
	}
	else_if = {
		limit = { has_idea = SZC_recovering_from_the_famine_1 }
		NOT = { has_idea = SZC_recovering_from_the_famine_1 }
	}
	else_if = {
		limit = { has_idea = SZC_recovering_from_the_famine_2 }
		NOT = { has_idea = SZC_recovering_from_the_famine_2 }
	}
}

SZC_opium_tier_1_unlocked = {
	if = {
		limit = { has_variable = SZC_opium_crisis_solved }
		hidden_trigger = { always = yes }
	}
	if = {
		limit = { has_variable = SZC_monopoly_path }
		custom_trigger_tooltip = {
			tooltip = SZC_opium_monopoly_tier_1
			check_variable = { SZC_opium_control > 0.19 }
		}
	}
	else_if = {
		limit = { has_variable = SZC_removal_path }
		custom_trigger_tooltip = {
			tooltip = SZC_opium_removal_tier_1
			check_variable = { SZC_opium_spread < 0.49 }
		}
	}
	else = {
		custom_trigger_tooltip = {
			tooltip = SZC_opium_crisis_solved
			has_variable = SZC_opium_crisis_solved
		}
	}
}
