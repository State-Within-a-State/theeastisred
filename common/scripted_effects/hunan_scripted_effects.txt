HNN_recruit_general_effect = {
	if = {
		limit = {
			NOT = {
				PREV = { tag = HNN }
			}
		}
		if = {
			limit = { THIS = { is_army_leader = yes } }
			if = {
				limit = { is_field_marshal = yes }
				demote_leader = yes
			}
			remove_trait = { trait = recently_promoted }
		}
		set_nationality = HNN
	}
}
