INR_add_rebel_units = {
	event_target:rebel_state = {
		create_unit = {
			division = "division_template = \"Rifle Division\" start_experience_factor = 0.5"
			owner = INR
		}
	}

	subtract_from_variable = { var = strength value = 1 }
	if = {
		limit = { check_variable = { strength > 0.99 } }
		INR_add_rebel_units = yes
	}
}
