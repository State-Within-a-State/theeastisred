add_namespace = asiapeace

country_event = { #Reichspakt is defeated in Asia
	id = asiapeace.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event asiapeace.1" }
	title = asiapeace.1.t
	desc = asiapeace.1.d

	is_triggered_only = yes

	trigger = {
		tag = GER
		has_war = yes
		is_faction_leader = yes
		GEA = {
			is_subject_of = ROOT
			has_capitulated = yes
		}
		any_enemy_country = {
			OR = {
				is_in_asia = yes
				is_in_australia = yes
			}
			has_capitulated = no
			OR = {
				is_faction_leader = yes
				is_in_faction = no
			}
		}
		NOT = {
			any_allied_country = {
				OR = {
					is_in_asia = yes
					is_in_australia = yes
				}
				has_capitulated = no
			}
		}
	}

	option = { #We surrender!
		name = asiapeace.1.a
		ai_chance = {
			factor = 100
		}
		custom_effect_tooltip = asian_reichspakt_will_proceed_to_peacedeal_we_white_peace
		hidden_effect = {
			# release all occupied states in Asia
			every_country = {
				limit = {
					has_war_with = GER
					OR = {
						tag = HAW
						is_in_asia = yes
						is_in_australia = yes
					}
				}
				every_owned_state = {
					limit = {
						OR = {
							is_controlled_by = GER
							controller = { is_in_faction_with = GER }
							controller = {
								NOT = { has_war_with = PREV.PREV }
							}
						}
					}
					owner = { set_state_controller = PREV }
				}
			}
			# remove Asian countries from the Reichspakt
			every_other_country = {
				limit = {
					is_ally_with = GER
					OR = {
						is_in_asia = yes
						is_in_australia = yes
						AND = {
							tag = HAW
							NOT = {
								USA = { is_in_faction_with = GER }
								TEX = { is_in_faction_with = GER }
							}
						}
					}
				}
				if = {
					limit = { is_subject = yes }
					overlord = {
						set_autonomy = {
							target = PREV
							autonomy_state = autonomy_free
						}
					}
				}
				leave_faction = yes
			}
			# then peace any Asian/Oceanian countries at war with the Reichspakt
			every_country = {
				limit = {
					is_subject = no
					has_war_with = GER
					OR = {
						is_in_asia = yes
						is_in_australia = yes
					}
					OR = {
						is_in_faction = no
						is_faction_leader = yes
					}
					NOT = {
						any_owned_state = {
							controller = { is_ally_with = GER }
						}
						any_allied_country = {
							any_owned_state = {
								controller = { is_ally_with = GER }
							}
						}
					}
				}
				set_country_flag = asian_adversary_for_germany
				country_event = asiapeace.2
			}
		}
	}

	option = { #We shall keep fighting
		name = asiapeace.1.b
		ai_chance = {
			factor = 0
		}
		add_war_support = -0.1
		add_stability = -0.1
	}
}

		hidden_effect = {
			if = {
				limit = {
					AUS = {
						has_war_together_with = GER
						has_war_with = JAP
					}
				}
				AUS = {
					white_peace_with_ROOT = yes
				}
			}
			GER = {
				white_peace_with_ROOT = yes
			}
		}
	}

	option = { #We demand unconditional surrender!
		name = asiapeace.2.b
		ai_chance = {
			factor = 0
		}
		add_war_support = -0.1
		add_stability = -0.1
		GER = {
			country_event = {
				id = asiapeace.3
				days = 1
			}
		}
	}
}
