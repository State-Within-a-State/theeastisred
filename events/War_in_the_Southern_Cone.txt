country_event = { #Argentina puppets Chile. Peru takes its claims
	id = southern.9
	title = southern.9.t
	desc = southern.9.d

	fire_only_once = yes

	trigger = {
		tag = ARG
		is_subject = no
		has_war_with = CHL
		OR = {
			is_in_faction = no
			is_faction_leader = yes
		}
		CHL = {
			surrender_progress > 0.8
			OR = {
				is_in_faction = no
				is_faction_leader = yes
			}
		}
	}

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event southern.9"
		hidden_effect = {
			CHL = {
				remove_ideas = {
					ARG_CHILE_STOP_BEING_A_TWAT
					CHL_home_of_south_american_syndicalism
				}
				clear_all_volunteers = yes
			}
		}
	}

	option = {
		ai_chance = {
			factor = 100
		}
		name = southern.9.a
		CHL = {
			### Transfer Peru's and Bolivia's claims, if at war
			if = {
				limit = {
					has_war_with = PRU
					has_war_with = BOL
				}
				every_owned_state = {
					limit = {
						OR = {
							state = 872 #Tacna
							state = 873 #Tarapacá
						}
					}
					PRU = { transfer_state = PREV }
				}
				BOL = { transfer_state = 506 } #Antofagasta
			}
			else = {
				every_owned_state = {
					limit = {
						OR = {
							state = 872 #Tacna
							state = 873 #Tarapacá
							state = 506 #Antofagasta
						}
					}
					random_country = {
						limit = {
							OR = {
								tag = PRU
								tag = BOL
							}
							has_war_with = ROOT
						}
						transfer_state = PREV
					}
				}
			}

			#Release FOP to prevent shenanigans
			set_autonomy = {
				target = FOP
				autonomy_state = autonomy_free
			}
			FOP = { country_event = southern.91 }

			#peace out Chile
			separate_peace_all_wars = yes

			### puppet remnants
			destroy_ships = {
				type = destroyer
				count = 4
			}
			destroy_ships = {
				type = light_cruiser
				count = 2
			}
			transfer_navy = { target = ROOT }
		}
		puppet = CHL
	}

	option = { #Annex Chile
		name = southern.9.b
		ai_chance = {
			factor = 0
		}
		add_stability = -0.1
		add_war_support = -0.1
		add_political_power = -150
		set_country_flag = CHL_CONQ
	}
}

country_event = { #Argentina takes Paraguay. Bolivia takes its claims
	id = southern.10
	title = southern.10.t
	desc = southern.10.d

	fire_only_once = yes

	trigger = {
		tag = ARG
		has_war_with = PAR
		is_subject = no
		surrender_progress < 0.30
		OR = {
			is_in_faction = no
			is_faction_leader = yes
		}
		PAR = {
			owns_state = 688 #Chaco Boreal
			surrender_progress > 0.90
			is_subject = no
		}
	}

	immediate = {
		log = "[GetDateText]: [Root.GetName]: event southern.10"
		#peace out Paraguay
		if = {
			limit = {
				688 = {
					is_owned_by = PAR
					is_controlled_by = BOL
				}
			}
			BOL = {
				transfer_state = 688
			}
		}
		PAR = {
			save_event_target_as = white_peace_receiver
		}
		ARG = {
			white_peace_with_TARGET = yes
		}
		BOL = {
			if = {
				limit = {
					exists = yes
					has_war_with = PAR
					OR = {
						is_in_faction = no
						is_faction_leader = yes
					}
				}
				white_peace_with_TARGET = yes
			}
		}

		#takes Paraguay and Chacos
		if = {
			limit = {
				1027 = {
					is_owned_by = PAR
				}
			}
			if = {
				limit = {
					country_exists = BOL
				}
				BOL = {
					transfer_state = 1027
				}
				else = {
					ARG = {
						transfer_state = 1027
					}
				}
			}
		}
		ARG = {
			transfer_state = 688
			if = {
				limit = {
					PAR = {
						owns_state = 301
					}
				}
				transfer_state = 301
			}
		}
	}

	option = { #Give them Chaco Boreal
		trigger = {
			country_exists = BOL
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 3
				has_global_flag = BOL_ARG
			}
		}
		name = southern.10.a
		BOL = {
			transfer_state = 688
		}
	}

	option = { #take Chaco Boreal
		ai_chance = {
			factor = 100
		}
		name = southern.10.b
		if = {
			limit = {
				BOL = {
					exists = yes
					is_subject = no
				}
			}
			BOL = {
				country_event = southern.11
			}
		}
	}
}
