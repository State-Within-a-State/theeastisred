	id = ire.242
	immediate = { log = "[GetDateText]: [Root.GetName]: event ire.241" }
	title = ire.242.t
	desc = ire.242.d
	picture = GFX_report_event_generic_riot
	fire_only_once = yes
	trigger = {
		tag = IRE
		has_country_flag = IRE_Ulster_Unlock_flag
		IRE_has_michael_collins = yes
	}

	option = { #Apply more Authoritarian Measures
		name = ire.242.a
		add_popularity = {
			ideology = authoritarian_democrat
			popularity = 0.12
		}
		set_country_flag = IRE_crackdown
	}
	option = { #Rig the Elections!
		name = ire.242.b
		if = {
			limit = { NOT = { has_government = paternal_autocrat } }
				add_popularity = {
				ideology = authoritarian_democrat
				popularity = 0.12
			}
			set_politics = {
				ruling_party = authoritarian_democrat
				elections_allowed = yes
			}
		}
		set_country_flag = IRE_crackdown
		
	}
	option = { #Continue the Crackdown
		name = ire.242.c
		set_country_flag = IRE_crackdown
	}
}
