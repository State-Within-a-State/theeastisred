	desc = italy.14.d

	trigger = {
		is_italian_federation = yes
		has_idea = Divided_Nation
		NOT = { country_exists = SRI }
		OR = {
			NOT = { owns_state = 742 } #Abruzzo
			NOT = { owns_state = 117 } #Campania
			NOT = { owns_state = 837 } #Puglia
			NOT = { owns_state = 156 } #Calabria
			NOT = { owns_state = 115 } #Sicilia
		}
	}

	fire_only_once = yes

	mean_time_to_happen = {
		days = 30
	}
	option = {
		name = italy.14.a
		remove_ideas = Divided_Nation

			limit = { is_core_of = SRI }
			remove_core_of = SRI
			if = {
				limit = { NOT = { is_core_of = ITA } }
				add_core_of = ITA
			}
		}
	}
