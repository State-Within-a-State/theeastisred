country_event = { #National Socialists Take Over in Japan
	id = rework_japan.3
	immediate = {log = "[GetDateText]: [Root.GetName]: event rework_japan.3"}
	title = rework_japan.3.title
	desc = rework_japan.3.desc
	picture = GFX_report_event_JAP_coup
	is_triggered_only = yes
	option = { #Japan will make a name for itself in the world.
		name = rework_japan.3.a
		ai_chance = { base = 100 }
		set_politics = {
			ruling_party = radical_socialist
			elections_allowed = no
		}
		add_popularity = {
			ideology = radical_socialist
			popularity = 0.3
		}
		set_party_name = {
			ideology = radical_socialist
			long_name = JAP_party_yuzonsha_long
			name = JAP_party_yuzonsha
		}
		retire_ideology_leader = radical_socialist
		create_country_leader = {
			name = "Kita Ikki"
			picture = "gfx/leaders/JAP/Portrait_JAP_Kita_Ikki.png"
			expire = "1965.1.1"
			ideology = fascism_subtype
		}
		hidden_effect = {
			remove_ideas = JAP_DEM_GOV
			set_country_flag = JAP_something_unexpected_happened
			remove_all_ministers = yes
			add_ideas = {
				JAP_Kita_Ikki_hog_rso
				JAP_Okawa_Shumei_for_npo
				JAP_Nishida_Mitsugi_eco_npo
				JAP_Shiono_Suehiko_sec_pau
			}
			if = {
				limit = { has_country_flag = JAP_yasuhito }
				add_ideas = JAP_emperor_yasuhito
			}
			else = { add_ideas = JAP_emperor_hirohito }
			mark_focus_tree_layout_dirty = yes
			every_country = {
				limit = { is_ai = no }
				set_variable = { superevents_numbers = 1 }
			}
			if = {
				limit = { NOT = { has_global_flag = KX_superevent_sounds_off } }
				sound_effect = SE_coup_of_kita_ikki_sounds
			}
			complete_national_focus = JAP_coup_of_national_socialists
		}
	}
}
