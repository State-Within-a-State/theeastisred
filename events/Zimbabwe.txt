country_event = {
	id = zim.13
	immediate = { log = "[GetDateText]: [Root.GetName]: event zim.13" }

	desc = zim.13.d

	is_triggered_only = yes
	trigger = {
		has_country_flag = saf_flag_elections_held
	}

	option = { #Rhodesia Party
		name = zim.13.a
		ai_chance = {
			factor = 30
		}
	
		if = {
			limit = {
				market_liberal < 0.5
			}

		}

		name = zim.13.b
		ai_chance = {
			factor = 50
		}
		
		if = {
			limit = {
				social_liberal < 0.5
			}

		}
		
		country_event = {
			id = rhodesia.12

		}

			limit = {
				has_country_flag = ZIM_huggins_compromised

			add_ideas = {
				ZIM_Thomas_Nangle_hog_sli

	option = { #Labour Party

			factor = 50
		}
		
		if = {
			limit = {
				social_democrat < 0.5
			}

		}

			ZIM_Harry_Davies_hog_sde
